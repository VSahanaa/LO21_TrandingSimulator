154 void Configuration::createSimulation() {
155  if(simulation) delete simulation;
156  dateDebut = ui->pickStartDate->date();
157  pourcentage = ui->pickPourcentage->value();
158  montantBaseInitial = ui->pickBase->value();
159  montantContrepartieInitial = ui->pickContrepartie->value();
160  EvolutionCours::iterator coursDebut = evolutionCours->searchCours(dateDebut);
161  if (modeSimulation == "Manuel") {
162  simulation = new ModeManuel(nomSimulation, evolutionCours, coursDebut, pourcentage, montantBaseInitial, montantContrepartieInitial);
163  }
164  else if (modeSimulation == "Pas_Pas") {
165  simulation = new ModePas_Pas(nomSimulation, evolutionCours, coursDebut, pourcentage, montantBaseInitial, montantContrepartieInitial);
166  }
167  else if (modeSimulation == "Automatique") {
168  QString strategieNom = ui->listeStrategie->currentText();
169  if(strategieNom == "MA Strategie") {
170  parameters.empty();
171  parameters.insert("period", ui->pickEmaPeriod->value());
172  }
173  else if (strategieNom == "RSI Strategie") {
174  parameters.empty();
175  parameters.insert("lookbackPeriod", ui->pickLookbackPeriod->value());
176  parameters.insert("overboughtBound", ui->pickOverbought->value());
177  parameters.insert("oversoldBound", ui->pickOversold->value());
178  }
179  if(strategie) delete strategie;
180  strategie = strategieFactory->getStrategie(strategieNom, evolutionCours);
181  strategie->setParameters(parameters);
182  simulation = new ModeAutomatique(nomSimulation, evolutionCours, coursDebut, strategie, pourcentage, montantBaseInitial, montantContrepartieInitial);
183  }
184  SimulationManager* simulationManager = SimulationManager::getSimulationManager();
185  simulationManager->addSimulation(simulation);
186  qDebug()<<simulation->getNom();
187 }
