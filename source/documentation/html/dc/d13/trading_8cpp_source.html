130 bool CoursOHLCV::isDragonflyDoji(double minLocal, double maxLocal) const {
131  /*
132  * Dragonfly Doji Formed when the opening and the closing prices are at the highest of the day.
133  * If it has a longer lower shadow it signals a more bullish trend. When appearing at market bottoms it is considered to be a reversal signal.
134  */
135  double mecheSup = high - max(open, close);
