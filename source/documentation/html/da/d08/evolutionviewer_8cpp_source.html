93 void EvolutionViewer::showChart(QDate firstdate, QDate lastdate) {
94  clearCharts();
95  EvolutionCours::iterator cours = evolutionCours->searchCours(firstdate);
96  qDebug() << firstdate.toString("yyyy");

98 
99  //add new data
100  Indicateur::iterator emaIterator, macdIterator;
101  QStringList dates;
102  dates << firstdate.toString("d/M");
103  int i = 0;
104  double yMin = cours->getLow();
105  double yMax = cours->getHigh();
106  for (; cours->getDate() <= lastdate; cours++){
107  if (cours->getHigh() > yMax) yMax = cours->getHigh(); //mesure range of y axis
108  if (cours->getLow() < yMin) yMin = cours->getLow();
109  Bougie* bougie = new Bougie(cours->getOpen(), cours->getHigh(), cours->getLow(), cours->getClose(), cours);

112  series->append(bougie);
113  //Indicateur series
114  if(EMA_series->isVisible()) {
115  emaIterator = ema->searchIndice(cours);
116  if(!emaIterator) {
117  EMA_series->append(i, 0);
118  }
119  else {
120  EMA_series->append(i, emaIterator->getIndice());
121  }
122  }
123  if(MACD_series->isVisible()) {
124  macdIterator = macd->searchIndice(cours);
125  int index = macd->index(macdIterator);
126  qDebug()<< "index " << index;
127  if(!macdIterator) {
128  MACD_series->append(i, 0);
129  MACD_signal->append(i,0);
130  MACD_histogram->append(i,0);
131  }
132  else {
133  MACD_series->append(i, macdIterator->getIndice());
134  MACD_signal->append(i, ((macd->signalLine_begin()) + index)->getIndice());
135  MACD_histogram->append(i, ((macd->histogramLine_begin()) + index)->getIndice());
136  }
137  }
138  i++;
139  dates << cours->getDate().toString("d/M");
140  if (cours == currentCours) break; //only show up to current cours
141  }
142  axisX->append(dates);
143  axisY->setRange(yMin*0.98, yMax*1.02);
144  chart->addSeries(series);
145  if(EMA_series->isVisible()) chart->addSeries(EMA_series);
146  if(MACD_series->isVisible()) {
147  chart->addSeries(MACD_series);
148  chart->addSeries(MACD_signal);
149  chart->addSeries(MACD_histogram);
150  }
151  chart->setAxisX(axisX,series);
152  chart->setAxisY(axisY,series);
153 }
