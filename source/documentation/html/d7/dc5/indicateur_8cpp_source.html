76 void RSI::generateIndice() {
77  //set parameters
78  nbMaxIndicateur = evolutionCours->getNbCours();
79  nbIndicateur = nbMaxIndicateur - lookbackPeriod -1;
80  delete [] indices;
81  indices = new IndiceIndicateur[nbMaxIndicateur];
82 
83  double avgGain=0, avgLost=0, RS;
84  EvolutionCours::iterator coursIterator;
85  double e;
86  for (coursIterator = evolutionCours->begin()+1; coursIterator != evolutionCours->begin()+lookbackPeriod+1; coursIterator++) {
87  e = coursIterator->getClose() - (coursIterator-1)->getClose();
88  if (e>0) {
89  avgGain += e;
90  }
91  else {
92  avgLost += -e;
93  }
94  }
95  avgGain /= lookbackPeriod;
96  avgLost /= lookbackPeriod;
97 
98  //il n'y a pas d'indicateur RSI pendant le look back periode
99  iterator indiceIterator = begin();
100 
101  //element suivant
102  while(indiceIterator != end()) {
103  e = coursIterator->getClose() - (coursIterator-1)->getClose();
104  if (e > 0) {
105  avgGain = (avgGain*(lookbackPeriod-1) + e)/lookbackPeriod;
106  }
107  else {
108  avgLost = (avgLost*(lookbackPeriod-1) - e)/lookbackPeriod;
109  }
110  RS = avgGain/avgLost;
111  indiceIterator->setDate(coursIterator->getDate());
112  indiceIterator->setIndice(100 - 100/(1+RS));
113  //qDebug() << indiceIterator->toString();
114  indiceIterator++;
115  coursIterator++;
116  }
