101 void MainInterface::updateTransactionTable() {
102  ui->transactionTable->setRowCount(0); //clear all data
103  TransactionManager::iterator transactionIterator = simulation->getTransactionManager()->head();
104  double montantTotalInit = simulation->getTransactionManager()->getMontantTotalInitial();
105  int i = 0;
106  while(transactionIterator != nullptr) {
107  ui->transactionTable->setRowCount(++i);
108  QTableWidgetItem *date, *diffContrepartie, *diffBase, *roi;
109  date = new QTableWidgetItem(transactionIterator->getCours()->getDate().toString("dd.MM.yy"), 0);
110  if(transactionIterator->differenceBase() > 0) {
111  diffBase = new QTableWidgetItem(QIcon(":/TradingSimulator/evolutionCours/icons/up.jpeg"), QString::number(transactionIterator->differenceBase()), 0);
112  }
113  else {
114  diffBase = new QTableWidgetItem(QIcon(":/TradingSimulator/evolutionCours/icons/down.jpeg"), QString::number(-transactionIterator->differenceBase()), 0);
115  }
116 
117  if(transactionIterator->differenceContrepartie() > 0) {
118  diffContrepartie = new QTableWidgetItem(QIcon(":/TradingSimulator/evolutionCours/icons/up.jpeg"), QString::number(transactionIterator->differenceContrepartie()), 0);
119  }
120  else {
121  diffContrepartie = new QTableWidgetItem(QIcon(":/TradingSimulator/evolutionCours/icons/down.jpeg"), QString::number(-transactionIterator->differenceContrepartie()), 0);
122  }
123  roi = new QTableWidgetItem(QString::number(transactionIterator->roi(montantTotalInit)), 0);
124  ui->transactionTable->setItem(i-1, 0, date);
125  ui->transactionTable->setItem(i-1, 1, diffContrepartie);
126  ui->transactionTable->setItem(i-1, 2, diffBase);
127  ui->transactionTable->setItem(i-1, 3, roi);
128  transactionIterator = transactionIterator->next();
129  }
130 }
